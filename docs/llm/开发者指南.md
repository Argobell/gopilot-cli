## 扩展 LLM 提供商

本指南将演示如何添加新的 LLM 提供商（如 Anthropic Claude、Google Gemini 等）。

### 步骤概览

1. 在 `src/schema.py` 中添加新的枚举值
2. 创建新的客户端类（继承 `LLMClientBase`）
3. 在 `LLMClient` 中添加工厂逻辑
4. 测试新提供商

#### 步骤 2：创建 Claude 客户端

```python
# src/llm/claude_client.py
from src.llm.base import LLMClientBase
from src.schema import Message, LLMResponse, ToolCall, FunctionCall
from anthropic import AsyncAnthropic
from typing import Any

class ClaudeClient(LLMClientBase):
    """Anthropic Claude 客户端实现"""

    def __init__(self, api_key: str, api_base: str, model: str):
        super().__init__(api_key, api_base, model)
        # 初始化 Anthropic 客户端
        self.client = AsyncAnthropic(api_key=api_key, base_url=api_base)

    async def generate(
        self,
        messages: list[Message],
        tools: list[Any] | None = None
    ) -> LLMResponse:
        """生成 Claude 响应"""

        # 转换为 Claude 格式
        claude_messages = self._convert_messages(messages)

        # 准备请求参数
        request_params = {
            "model": self.model,
            "max_tokens": 1024,
            "messages": claude_messages,
        }

        # 添加工具（如果存在）
        if tools:
            # 转换为 Claude 格式
            claude_tools = self._convert_tools(tools)
            request_params["tools"] = claude_tools

        # 调用 Claude API
        response = await self.client.messages.create(**request_params)

        # 解析响应
        return self._parse_response(response)

    def _convert_messages(self, messages: list[Message]) -> list[dict[str, Any]]:
        """转换消息为 Claude 格式"""
        claude_messages = []

        for msg in messages:
            if msg.role == "system":
                # Claude 使用 system 字段
                claude_messages.append({
                    "role": "system",
                    "content": msg.content
                })
            elif msg.role == "user":
                claude_messages.append({
                    "role": "user",
                    "content": msg.content
                })
            elif msg.role == "assistant":
                claude_messages.append({
                    "role": "assistant",
                    "content": msg.content
                })

        return claude_messages

    def _convert_tools(self, tools: list[Any]) -> list[dict[str, Any]]:
        """转换工具为 Claude 格式"""

        claude_tools = []

        for tool in tools:
            # OpenAI 格式 → Claude 格式
            if isinstance(tool, dict) and tool.get("type") == "function":
                claude_tools.append({
                    "name": tool["function"]["name"],
                    "description": tool["function"]["description"],
                    "input_schema": tool["function"]["parameters"]
                })
            elif hasattr(tool, "to_claude_schema"):
                # 自定义对象
                claude_tools.append(tool.to_claude_schema())

        return claude_tools

    def _parse_response(self, response) -> LLMResponse:
        """解析 Claude 响应"""
        content = response.content[0].text

        # 提取工具调用（Claude 使用不同的方式）
        tool_calls = None
        if hasattr(response, 'tool_calls') and response.tool_calls:
            tool_calls = [
                ToolCall(
                    id=tc.id,
                    type="function",
                    function=FunctionCall(
                        name=tc.function.name,
                        arguments=tc.function.arguments
                    )
                )
                for tc in response.tool_calls
            ]

        return LLMResponse(
            content=content,
            thinking=None,  # Claude 不支持 reasoning_content
            tool_calls=tool_calls,
            finish_reason=response.stop_reason
        )
```

## 自定义消息转换器

有时您可能需要自定义消息格式转换逻辑，以支持特定的 LLM 服务。

### 场景 1：自定义角色名称

某些服务可能使用不同的角色名称（如 `"human"` vs `"user"`）。

```python
class CustomOpenAIClient(OpenAIClient):
    """自定义消息格式的 OpenAI 客户端"""

    def _convert_messages(self, messages: list[Message]) -> tuple[str | None, list[dict[str, Any]]]:
        """自定义消息转换"""

        system_message = None
        api_messages = []

        for msg in messages:
            if msg.role == "system":
                system_message = msg.content
            elif msg.role == "user":
                # 自定义：使用 "human" 而不是 "user"
                api_messages.append({
                    "role": "human",
                    "content": msg.content
                })
            elif msg.role == "assistant":
                api_messages.append({
                    "role": "assistant",
                    "content": msg.content
                })

        return system_message, api_messages
```

```python
class ThinkingClient(OpenAIClient):
    """自定义思考内容格式"""

                    "role": msg.role,
                    "content": msg.content
                })

        return system_message, api_messages
```

```python
class SafeClient(OpenAIClient):
    """过滤敏感信息的客户端"""

    def _convert_messages(self, messages: list[Message]) -> tuple[str | None, list[dict[str, Any]]]:
        """过滤敏感内容"""

        filtered_messages = []

        for msg in messages:
            # 过滤敏感词
            filtered_content = self._filter_sensitive_words(msg.content)
            filtered_msg = msg.model_copy(update={"content": filtered_content})
            filtered_messages.append(filtered_msg)

        return super()._convert_messages(filtered_messages)

    def _filter_sensitive_words(self, text: str) -> str:
        """简单的敏感词过滤示例"""
        sensitive_words = ["密码", "信用卡", "身份证"]
        for word in sensitive_words:
            text = text.replace(word, "***")
        return text
```

## 自定义工具格式

### 场景 1：创建工具类

```python
from typing import Any, Dict
from pydantic import BaseModel

class BaseTool:
    """所有工具的基类"""

    @property
    def name(self) -> str:
        """工具名称"""
        raise NotImplementedError

    @property
    def description(self) -> str:
        """工具描述"""
        raise NotImplementedError

    @property
    def parameters(self) -> Dict[str, Any]:
        """参数模式"""
        raise NotImplementedError

    def to_openai_schema(self) -> Dict[str, Any]:
        """转换为 OpenAI 格式"""
        return {
            "type": "function",
            "function": {
                "name": self.name,
                "description": self.description,
                "parameters": self.parameters
            }
        }

    def __call__(self, **kwargs) -> Any:
        """执行工具逻辑"""
        raise NotImplementedError

# 示例：天气工具
class WeatherTool(BaseTool):
    """天气查询工具"""

    @property
    def name(self) -> str:
        return "get_weather"

    @property
    def description(self) -> str:
        return "获取指定城市的天气信息"

    @property
    def parameters(self) -> Dict[str, Any]:
        return {
            "type": "object",
            "properties": {
                "city": {
                    "type": "string",
                    "description": "城市名称，如：北京、上海"
                },
                "unit": {
                    "type": "string",
                    "enum": ["celsius", "fahrenheit"],
                    "description": "温度单位，默认 celsius"
                }
            },
            "required": ["city"]
        }

    def __call__(self, city: str, unit: str = "celsius") -> Dict[str, Any]:
        """执行天气查询"""
        # 这里可以调用真实的天气 API
        return {
            "city": city,
            "weather": "晴天",
            "temperature": 25,
            "unit": unit
        }

# 使用
weather_tool = WeatherTool()
tools = [weather_tool.to_openai_schema()]

response = await client.generate(messages, tools=tools)

# 执行工具
if response.tool_calls:
```

### 场景 2：工具组合

```python
class ToolRegistry:
    """工具注册表"""

    def __init__(self):
        self._tools: Dict[str, BaseTool] = {}

    def register(self, tool: BaseTool):
        """注册工具"""
        self._tools[tool.name] = tool

    def get_tool(self, name: str) -> BaseTool:
        """获取工具"""
        if name not in self._tools:
            raise ValueError(f"工具未注册: {name}")
        return self._tools[name]

    def to_openai_schemas(self) -> list[Dict[str, Any]]:
        """转换为 OpenAI 工具格式"""
        return [tool.to_openai_schema() for tool in self._tools.values()]

    def execute_tool(self, name: str, arguments: str) -> Dict[str, Any]:
        """执行工具"""
        import json
        tool = self.get_tool(name)
        args = json.loads(arguments)
        return tool(**args)

# 使用注册表
registry = ToolRegistry()
registry.register(WeatherTool())
registry.register(DatabaseTool())
registry.register(EmailTool())

# 获取所有工具
tools = registry.to_openai_schemas()

# 使用
response = await client.generate(messages, tools=tools)

# 执行工具
if response.tool_calls:
    for tool_call in response.tool_calls:
        result = registry.execute_tool(
            tool_call.function.name,
            tool_call.function.arguments
        )
```

### 场景 3：工具链调用

```python
class ToolChain:
    """工具链：多个工具按顺序执行"""

    def __init__(self, registry: ToolRegistry):
        self.registry = registry
        self.history: list[Dict[str, Any]] = []

    async def execute(self, messages: list[Message], tools: list[Dict[str, Any]]) -> LLMResponse:
        """执行工具链"""

        # 初始响应
        response = await client.generate(messages, tools=tools)
        current_messages = messages + [Message(role="assistant", content=response.content)]

        # 循环处理工具调用
        max_iterations = 5
        iteration = 0

        while response.tool_calls and iteration < max_iterations:
            iteration += 1

            # 执行所有工具调用
            for tool_call in response.tool_calls:
                result = self.registry.execute_tool(
                    tool_call.function.name,
                    tool_call.function.arguments
                )

                # 记录工具执行历史
                self.history.append({
                    "tool": tool_call.function.name,
                    "arguments": tool_call.function.arguments,
                    "result": result
                })

                # 添加工具结果到消息
                current_messages.append(Message(
                    role="tool",
                    content=str(result),
                    tool_call_id=tool_call.id,
                    name=tool_call.function.name
                ))

            # 继续生成（包含工具结果）
            response = await client.generate(current_messages, tools=tools)
            current_messages.append(Message(role="assistant", content=response.content))

        return response

# 使用
chain = ToolChain(registry)
final_response = await chain.execute(messages, tools)
```

```python
import json
import jsonschema

def execute_tool_safely(tool_call: ToolCall, tools: list[BaseTool]):
    """安全执行工具"""

    # 1. 查找工具
    tool = next((t for t in tools if t.name == tool_call.function.name), None)
    if not tool:
        raise ValueError(f"工具不存在: {tool_call.function.name}")

    # 2. 解析参数
    try:
        args = json.loads(tool_call.function.arguments)
    except json.JSONDecodeError as e:
        raise ValueError(f"参数格式错误: {e}")

    # 3. 验证参数
    try:
        jsonschema.validate(args, tool.parameters)
    except jsonschema.ValidationError as e:
        raise ValueError(f"参数验证失败: {e}")

    # 4. 执行工具
    return tool(**args)
```

### 5. 错误处理

#### ✅ 推荐：使用重试机制

```python
import asyncio
from asyncio import timeout

async def retry_with_backoff(
    func,
    max_retries: int = 3,
    base_delay: float = 1.0
):
    """带指数退避的重试"""

    for attempt in range(max_retries):
        try:
            return await func()
        except Exception as e:
            if attempt == max_retries - 1:
                raise

            delay = base_delay * (2 ** attempt)
            await asyncio.sleep(delay)
```

#### ✅ 推荐：分类处理异常

```python
async def handle_error(e: Exception):
    """错误分类处理"""

    if isinstance(e, ConnectionError):
        # 连接错误：重试或检查网络
        return "连接失败，请检查网络或服务状态"
    elif isinstance(e, AuthenticationError):
        # 认证错误：检查 API 密钥
        return "认证失败，请检查 API 密钥"
    elif isinstance(e, ValueError):
        # 参数错误：检查输入
        return f"参数错误: {e}"
    else:
        # 其他错误：记录并报告
        return f"未知错误: {type(e).__name__}"
```

## 错误处理策略

### 完整示例：错误处理和重试

```python
import asyncio
import logging
from typing import Optional

# 配置日志
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class LLMError(Exception):
    """LLM 异常基类"""
    pass

class ConnectionError(LLMError):
    """连接错误"""
    pass

class AuthenticationError(LLMError):
    """认证错误"""
    pass

class RateLimitError(LLMError):
    """速率限制错误"""
    pass

class RobustLLMClient:
    """带完整错误处理的 LLM 客户端"""

    def __init__(self, client: LLMClient):
        self.client = client

    async def generate_with_retry(
        self,
        messages: list[Message],
        tools: Optional[list] = None,
        max_retries: int = 3,
        timeout_seconds: int = 60
    ) -> Optional[LLMResponse]:
        """带重试和超时的生成"""

        for attempt in range(max_retries):
            try:
                # 使用超时
                async with asyncio.timeout(timeout_seconds):
                    response = await self.client.generate(messages, tools)
                    logger.info(f"请求成功 (尝试 {attempt + 1})")
                    return response

            except asyncio.TimeoutError:
                logger.warning(f"超时 (尝试 {attempt + 1}/{max_retries})")
                if attempt == max_retries - 1:
                    raise ConnectionError(f"请求超时 ({timeout_seconds}秒)")

            except ConnectionError as e:
                logger.warning(f"连接错误 (尝试 {attempt + 1}/{max_retries}): {e}")
                if attempt == max_retries - 1:
                    raise

            except AuthenticationError as e:
                logger.error(f"认证失败: {e}")
                raise  # 不重试认证错误

            except RateLimitError as e:
                logger.warning(f"速率限制 (尝试 {attempt + 1}/{max_retries}): {e}")
                # 速率限制：使用更长的延迟
                delay = 2 ** (attempt + 1)
                await asyncio.sleep(delay)

            except Exception as e:
                logger.error(f"未知错误 (尝试 {attempt + 1}/{max_retries}): {type(e).__name__}: {e}")
                if attempt == max_retries - 1:
                    raise LLMError(f"请求失败: {e}")

            # 重试延迟
            if attempt < max_retries - 1:
                delay = 2 ** attempt
                logger.info(f"等待 {delay} 秒后重试...")
                await asyncio.sleep(delay)

        return None

# 使用示例
async def main():
    client = LLMClient(api_key="your-key", ...)
    robust_client = RobustLLMClient(client)

    try:
        response = await robust_client.generate_with_retry([
            Message(role="user", content="你好")
        ])

        if response:
            print(response.content)
        else:
            print("请求失败，已达到最大重试次数")

    except LLMError as e:
        print(f"LLM 错误: {e}")
    except Exception as e:
        print(f"系统错误: {type(e).__name__}: {e}")

asyncio.run(main())
```

```python
class SingletonAsyncOpenAI:
    """单例 AsyncOpenAI 客户端"""

    _instances: dict[str, AsyncOpenAI] = {}

    @classmethod
    def get_client(
        cls,
        api_key: str,
        api_base: str
    ) -> AsyncOpenAI:
        """获取或创建客户端"""
        key = f"{api_key}:{api_base}"
        if key not in cls._instances:
            cls._instances[key] = AsyncOpenAI(
                api_key=api_key,
                base_url=api_base
            )
        return cls._instances[key]
```

### 2. 并发请求

```python
async def batch_generate(
    client: LLMClient,
    requests: list[list[Message]]
) -> list[LLMResponse]:
    """批量并发生成"""

    async def generate_single(request):
        return await client.generate(request)

    # 并发执行所有请求
    tasks = [generate_single(req) for req in requests]
    responses = await asyncio.gather(*tasks)

    return responses

# 使用
requests = [
    [Message(role="user", content=f"问题 {i}")]
    for i in range(10)
]

responses = await batch_generate(client, requests)
```

### 3. 缓存结果

```python
import hashlib
import json
from functools import lru_cache

class CachedLLMClient:
    """带缓存的 LLM 客户端"""

    def __init__(self, client: LLMClient):
        self.client = client
        self._cache: dict[str, LLMResponse] = {}

    def _get_cache_key(self, messages: list[Message]) -> str:
        """生成缓存键"""
        content = json.dumps([msg.model_dump() for msg in messages], sort_keys=True)
        return hashlib.md5(content.encode()).hexdigest()

    async def generate(
        self,
        messages: list[Message],
        tools: Optional[list] = None
    ) -> LLMResponse:
        """带缓存的生成"""

        cache_key = self._get_cache_key(messages)

        if cache_key in self._cache:
            return self._cache[cache_key]

        response = await self.client.generate(messages, tools)
        self._cache[cache_key] = response

        return response

    def clear_cache(self):
        """清空缓存"""
        self._cache.clear()
```
