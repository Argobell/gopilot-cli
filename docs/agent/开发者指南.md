## 代码结构

### 核心文件

```
src/agent.py
├── Colors 类 (第 15-39 行)
│   └── ANSI 颜色定义
│
└── Agent 类 (第 41-401 行)
    ├── 初始化 (__init__, 第 44-69 行)
    ├── 消息管理 (add_user_message, 第 71-73 行)
    ├── Token 估算 (_estimate_tokens, 第 75-126 行)
    ├── 消息摘要 (_summarize_messages, 第 127-199 行)
    ├── 摘要创建 (_create_summary, 第 200-266 行)
    └── 执行循环 (run, 第 267-401 行)
```

### 关键设计点

**1. 单一职责原则**
- `Colors`: 只负责颜色定义
- `Agent`: 只负责协调和执行

**2. 依赖倒置**
- Agent 依赖 `LLMClient` 抽象，不依赖具体实现
- 通过工具接口实现可插拔

**3. 开闭原则**
- 对扩展开放：可以添加新工具、新消息处理器
- 对修改关闭：核心逻辑稳定

## 扩展 Agent

### 方式 1：继承 Agent 类

```python
from src.agent import Agent, Colors
from src.llm import LLMClient
from src.tools.base import Tool, ToolResult
from src.schema import Message

class CustomAgent(Agent):
    """自定义 Agent 示例"""

    def __init__(self, llm_client: LLMClient, **kwargs):
        # 调用父类初始化
        super().__init__(
            llm_client=llm_client,
            system_prompt="你是一个自定义助手",
            tools=[],
            **kwargs
        )

        # 添加自定义属性
        self.execution_log = []

    async def run(self):
        """重写执行逻辑"""
        print(f"{Colors.BRIGHT_CYAN}开始自定义执行...{Colors.RESET}")

        # 添加自定义逻辑
        await self._log_execution()

        # 调用父类逻辑
        result = await super().run()

        # 添加后处理
        await self._finalize_execution()

        return result

    async def _log_execution(self):
        """记录执行日志"""
        self.execution_log.append({
            "action": "execution_start",
            "timestamp": asyncio.get_event_loop().time()
        })

    async def _finalize_execution(self):
        """最终化执行"""
        self.execution_log.append({
            "action": "execution_end",
            "timestamp": asyncio.get_event_loop().time()
        })

# 使用
async def custom_agent_example():
    agent = CustomAgent(
        llm_client=llm_client,
        max_steps=10
    )

    agent.add_user_message("你好")
    result = await agent.run()

    print(f"执行日志: {agent.execution_log}")

# 运行
# asyncio.run(custom_agent_example())
```

### 方式 2：组合模式（推荐）

```python
class AgentExtension:
    """Agent 扩展基类"""

    def before_run(self, agent: Agent):
        """run() 执行前的钩子"""
        pass

    def after_run(self, agent: Agent, result: str):
        """run() 执行后的钩子"""
        pass

    def on_tool_call(self, agent: Agent, tool_call):
        """工具调用时的钩子"""
        pass

    def on_error(self, agent: Agent, error: Exception):
        """错误发生时的钩子"""
        pass

class LoggingExtension(AgentExtension):
    """日志扩展"""

    def __init__(self):
        self.logs = []

    def before_run(self, agent: Agent):
        self.logs.append("开始执行")

    def after_run(self, agent: Agent, result: str):
        self.logs.append(f"执行完成: {result[:50]}...")

    def on_tool_call(self, agent: Agent, tool_call):
        self.logs.append(f"调用工具: {tool_call.function.name}")

class AgentWithExtensions(Agent):
    """支持扩展的 Agent"""

    def __init__(self, llm_client: LLMClient, extensions: list[AgentExtension] = None, **kwargs):
        super().__init__(llm_client, **kwargs)
        self.extensions = extensions or []

    async def run(self) -> str:
        # 执行前的钩子
        for ext in self.extensions:
            ext.before_run(self)

        try:
            result = await super().run()
            return result
        except Exception as e:
            for ext in self.extensions:
                ext.on_error(self, e)
            raise
        finally:
            # 执行后的钩子
            for ext in self.extensions:
                ext.after_run(self, result)

# 使用
async def extension_example():
    extensions = [LoggingExtension()]

    agent = AgentWithExtensions(
        llm_client=llm_client,
        extensions=extensions,
        tools=[MyTool()],
        max_steps=10
    )

    agent.add_user_message("测试扩展")
    result = await agent.run()

    print(extensions[0].logs)

# 运行
# asyncio.run(extension_example())
```

### 方式 3：包装器模式

```python
def with_timeout(timeout_seconds: int):
    """添加超时功能的包装器"""

    def decorator(func):
        async def wrapper(agent_self, *args, **kwargs):
            try:
                result = await asyncio.wait_for(
                    func(agent_self, *args, **kwargs),
                    timeout=timeout_seconds
                )
                return result
            except asyncio.TimeoutError:
                print(f"{Colors.BRIGHT_RED}执行超时！{Colors.RESET}")
                return "任务执行超时"

        return wrapper
    return decorator

class TimeoutAgent(Agent):
    """带超时的 Agent"""

    @with_timeout(30)  # 30 秒超时
    async def run(self):
        return await super().run()

# 使用
async def timeout_example():
    agent = TimeoutAgent(
        llm_client=llm_client,
        system_prompt="你是一个助手",
        tools=[],
        max_steps=100  # 大量步骤可能超时
    )

    agent.add_user_message("执行一个长时间的任务")
    result = await agent.run()

    print(f"结果: {result}")
```

```python
                content="",
```

### 实际示例 1：文件操作工具

```python
import os
from pathlib import Path
from src.tools.base import Tool, ToolResult

class FileOperationTool(Tool):
    """文件操作工具"""

    name = "file_operation"
    description = "执行文件读写操作"

    async def execute(
        self,
        operation: str,
        file_path: str,
        content: str = None,
        encoding: str = "utf-8"
    ) -> ToolResult:
        """
        文件操作

        Args:
            operation: 操作类型 (read, write, list, delete)
            file_path: 文件路径
            content: 写入内容（write 操作需要）
            encoding: 文件编码

        Returns:
            操作结果
        """
        try:
            file_path = Path(file_path)

            # 验证操作类型
            valid_operations = ["read", "write", "list", "delete"]
            if operation not in valid_operations:
                return ToolResult(
                    success=False,
                    content="",
                    error=f"无效操作: {operation}，有效值: {valid_operations}"
                )

            # 执行对应操作
            if operation == "read":
                return await self._read_file(file_path, encoding)

            elif operation == "write":
                if content is None:
                    return ToolResult(
                        success=False,
                        content="",
                        error="write 操作需要提供 content 参数"
                    )
                return await self._write_file(file_path, content, encoding)

            elif operation == "list":
                return await self._list_directory(file_path)

            elif operation == "delete":
                return await self._delete_file(file_path)

        except Exception as e:
            return ToolResult(
                success=False,
                content="",
                error=f"文件操作失败: {str(e)}"
            )

    async def _read_file(self, file_path: Path, encoding: str) -> ToolResult:
        """读取文件"""
        if not file_path.exists():
            return ToolResult(
                success=False,
                content="",
                error=f"文件不存在: {file_path}"
            )

        if not file_path.is_file():
            return ToolResult(
                success=False,
                content="",
                error=f"不是文件: {file_path}"
            )

        content = file_path.read_text(encoding=encoding)
        return ToolResult(
            success=True,
            content=f"文件内容 ({file_path}):\n{content}",
            error=""
        )

    async def _write_file(self, file_path: Path, content: str, encoding: str) -> ToolResult:
        """写入文件"""
        # 创建目录（如果不存在）
        file_path.parent.mkdir(parents=True, exist_ok=True)

        file_path.write_text(content, encoding=encoding)
        return ToolResult(
            success=True,
            content=f"文件已写入: {file_path}",
            error=""
        )

    async def _list_directory(self, file_path: Path) -> ToolResult:
        """列出目录内容"""
        if not file_path.exists():
            return ToolResult(
                success=False,
                content="",
                error=f"目录不存在: {file_path}"
            )

        if not file_path.is_dir():
            return ToolResult(
                success=False,
                content="",
                error=f"不是目录: {file_path}"
            )

        items = list(file_path.iterdir())
        result = f"目录内容 ({file_path}):\n"
        for item in items:
            result += f"  - {item.name}\n"

        return ToolResult(success=True, content=result, error="")

    async def _delete_file(self, file_path: Path) -> ToolResult:
        """删除文件"""
        if not file_path.exists():
            return ToolResult(
                success=False,
                content="",
                error=f"文件不存在: {file_path}"
            )

        file_path.unlink()
        return ToolResult(
            success=True,
            content=f"文件已删除: {file_path}",
            error=""
        )

# 使用
async def file_tool_example():
    tool = FileOperationTool()

    # 写入文件
    result = await tool.execute(
        operation="write",
        file_path="/tmp/test.txt",
        content="Hello, World!"
    )
    print(result.content)

    # 读取文件
    result = await tool.execute(
        operation="read",
        file_path="/tmp/test.txt"
    )
    print(result.content)

# 运行
# asyncio.run(file_tool_example())
```

```python
import aiohttp
from src.tools.base import Tool, ToolResult

class HttpRequestTool(Tool):
    """HTTP 请求工具"""

    name = "http_request"
    description = "发送 HTTP 请求获取数据"

    async def execute(
        self,
        url: str,
        method: str = "GET",
        headers: dict = None,
        data: str = None,
        timeout: int = 30
    ) -> ToolResult:
        """
        发送 HTTP 请求

        Args:
            url: 请求 URL
            method: HTTP 方法 (GET, POST, PUT, DELETE)
            headers: 请求头
            data: 请求体（POST/PUT 需要）
            timeout: 超时时间（秒）

        Returns:
            响应内容
        """
        try:
            headers = headers or {}

            async with aiohttp.ClientSession() as session:
                async with session.request(
                    method=method.upper(),
                    url=url,
                    headers=headers,
                    data=data,
                    timeout=aiohttp.ClientTimeout(total=timeout)
                ) as response:

                    # 获取响应内容
                    if "application/json" in response.headers.get("Content-Type", ""):
                        content = await response.json()
                        content_str = str(content)
                    else:
                        content_str = await response.text()

                    # 返回结果
                    return ToolResult(
                        success=True,
                        content=(
                            f"状态码: {response.status}\n"
                            f"URL: {url}\n"
                            f"响应内容:\n{content_str[:500]}"  # 限制长度
                        ),
                        error=""
                    )

        except asyncio.TimeoutError:
            return ToolResult(
                success=False,
                content="",
                error=f"请求超时 ({timeout} 秒)"
            )
        except Exception as e:
            return ToolResult(
                success=False,
                content="",
                error=f"请求失败: {str(e)}"
            )

# 使用
async def http_tool_example():
    tool = HttpRequestTool()

    result = await tool.execute(
        url="https://httpbin.org/get",
        method="GET"
    )
    print(result.content)

# 运行
# asyncio.run(http_tool_example())
```

### 实际示例 3：数据分析工具

```python
import csv
import json
from io import StringIO
from typing import List, Dict, Any
from src.tools.base import Tool, ToolResult

class DataAnalysisTool(Tool):
    """数据分析工具"""

    name = "analyze_data"
    description = "分析 CSV/JSON 数据，计算统计指标"

    async def execute(
        self,
        data: str,
        format: str = "csv",
        operation: str = "summary",
        column: str = None
    ) -> ToolResult:
        """
        数据分析

        Args:
            data: 数据内容（CSV 或 JSON）
            format: 数据格式 (csv, json)
            operation: 操作类型 (summary, filter, aggregate, sort)
            column: 列名（某些操作需要）

        Returns:
            分析结果
        """
        try:
            if format == "csv":
                return await self._analyze_csv(data, operation, column)
            elif format == "json":
                return await self._analyze_json(data, operation, column)
            else:
                return ToolResult(
                    success=False,
                    content="",
                    error=f"不支持的格式: {format}，支持: csv, json"
                )

        except Exception as e:
            return ToolResult(
                success=False,
                content="",
                error=f"数据分析失败: {str(e)}"
            )

    async def _analyze_csv(self, data: str, operation: str, column: str) -> ToolResult:
        """分析 CSV 数据"""
        # 解析 CSV
        csv_reader = csv.DictReader(StringIO(data))
        rows = list(csv_reader)

        if not rows:
            return ToolResult(success=True, content="数据为空", error="")

        if operation == "summary":
            return await self._csv_summary(rows)
        elif operation == "filter":
            if not column:
                return ToolResult(
                    success=False,
                    content="",
                    error="filter 操作需要指定 column"
                )
            return await self._csv_filter(rows, column)
        elif operation == "aggregate":
            if not column:
                return ToolResult(
                    success=False,
                    content="",
                    error="aggregate 操作需要指定 column"
                )
            return await self._csv_aggregate(rows, column)
        else:
            return ToolResult(
                success=False,
                content="",
                error=f"不支持的操作: {operation}"
            )

    async def _csv_summary(self, rows: List[Dict[str, str]]) -> ToolResult:
        """CSV 数据摘要"""
        num_rows = len(rows)
        num_cols = len(rows[0].keys()) if rows else 0

        # 提取数值列
        numeric_cols = []
        if rows:
            for col in rows[0].keys():
                try:
                    float(list(rows[0].values())[0])
                    numeric_cols.append(col)
                except (ValueError, IndexError):
                    pass

        result = f"数据摘要:\n"
        result += f"  行数: {num_rows}\n"
        result += f"  列数: {num_cols}\n"
        result += f"  列名: {', '.join(rows[0].keys())}\n"
        result += f"  数值列: {', '.join(numeric_cols) if numeric_cols else '无'}\n"

        # 数值列统计
        if numeric_cols:
            result += "\n数值列统计:\n"
            for col in numeric_cols:
                values = [float(row[col]) for row in rows if row[col]]
                if values:
                    result += f"  {col}:\n"
                    result += f"    平均值: {sum(values) / len(values):.2f}\n"
                    result += f"    最小值: {min(values)}\n"
                    result += f"    最大值: {max(values)}\n"

        return ToolResult(success=True, content=result, error="")

    async def _csv_filter(self, rows: List[Dict[str, str]], column: str) -> ToolResult:
        """筛选 CSV 数据"""
        # 简单筛选：返回唯一值
        values = list({row[column] for row in rows if column in row})
        result = f"列 '{column}' 的唯一值:\n" + "\n".join(f"  - {v}" for v in values)
        return ToolResult(success=True, content=result, error="")

    async def _csv_aggregate(self, rows: List[Dict[str, str]], column: str) -> ToolResult:
        """聚合 CSV 数据"""
        if column not in rows[0]:
            return ToolResult(
                success=False,
                content="",
                error=f"列 '{column}' 不存在"
            )

        # 计算数值列的平均值
        values = []
        for row in rows:
            if row[column]:
                try:
                    values.append(float(row[column]))
                except ValueError:
                    pass

        if not values:
            return ToolResult(success=True, content=f"列 '{column}' 无数值数据", error="")

        avg = sum(values) / len(values)
        result = f"列 '{column}' 聚合结果:\n  平均值: {avg:.2f}\n  总数: {len(values)}"
        return ToolResult(success=True, content=result, error="")

    async def _analyze_json(self, data: str, operation: str, column: str) -> ToolResult:
        """分析 JSON 数据"""
        try:
            json_data = json.loads(data)
            result = f"JSON 数据:\n{json.dumps(json_data, indent=2, ensure_ascii=False)[:500]}"
            return ToolResult(success=True, content=result, error="")
        except json.JSONDecodeError as e:
            return ToolResult(
                success=False,
                content="",
                error=f"JSON 解析失败: {str(e)}"
            )

# 使用
async def data_analysis_example():
    tool = DataAnalysisTool()

    # 分析 CSV 数据
    csv_data = """name,age,score
Alice,25,95
Bob,30,87
Charlie,28,92"""

    result = await tool.execute(
        data=csv_data,
        format="csv",
        operation="summary"
    )
    print(result.content)

# 运行
```

**3. 完善的错误处理**

```python
async def execute(self, file_path: str) -> ToolResult:
    # 1. 参数验证
    if not file_path:
        return ToolResult(
            success=False,
            content="",
            error="file_path 参数不能为空"
        )

    # 2. 文件存在性检查
    if not os.path.exists(file_path):
        return ToolResult(
            success=False,
            content="",
            error=f"文件不存在: {file_path}"
        )

    # 3. 权限检查
    if not os.access(file_path, os.R_OK):
        return ToolResult(
            success=False,
            content="",
            error=f"没有文件读取权限: {file_path}"
        )

    # 4. 执行操作
    try:
        content = self._read_file(file_path)
        return ToolResult(success=True, content=content, error="")
```

## 贡献指南

### 提交流程

**1. Fork 仓库**

```bash
# 在 GitHub 上 Fork https://github.com/Argobell/gopilot-cli
git clone https://github.com/your-username/gopilot-cli.git
cd gopilot-cli
```

**2. 创建特性分支**

```bash
git checkout -b feature/your-feature-name
```

**3. 提交代码**

```bash
git add .
git commit -m "feat: 添加新功能描述"
```

**4. 推送并创建 PR**

```bash
git push origin feature/your-feature-name
# 在 GitHub 上创建 Pull Request
```

### 代码规范

**提交信息格式**

```
<type>(<scope>): <subject>

<body>

<footer>
```

**类型（type）**
- `feat`: 新功能
- `fix`: 修复 bug
- `docs`: 文档更新
- `style`: 代码格式
- `refactor`: 代码重构
- `test`: 测试
- `chore`: 构建/工具相关

**示例**

```bash
feat(agent): 添加消息摘要功能

- 实现智能消息摘要机制
- 防止 Token 溢出
- 保留关键用户信息

Closes #123
```

**代码风格**

- 遵循 PEP 8
- 使用 Black 格式化：`black src/`
- 使用 isort 排序导入：`isort src/`
- 类型注解完整
- 文档字符串完整

### PR 要求

**必要检查**

1. **代码质量**
   - 通过所有测试
   - 通过 linting 检查
   - 类型检查通过

2. **测试覆盖**
   - 新功能必须有测试
   - 修复 bug 需要回归测试
   - 测试覆盖率 > 80%

3. **文档**
   - 更新相关文档
   - 添加 API 文档
   - 示例代码

**PR 模板**

```markdown
## 变更说明
简要描述本次变更
```

## 常见问题

### Q1: 如何添加新的 LLM 提供商？

**A:** 查看 [LLM 模块文档](../llm/README.md) 了解如何扩展 LLM 客户端。

### Q2: 工具返回大结果时被截断怎么办？

**A:** 检查 `Agent.run()` 方法中的结果截断逻辑（第 374-377 行）：

```python
if len(result_text) > 300:
    result_text = result_text[:300] + f"{Colors.DIM}...{Colors.RESET}"
```

修改这个限制值或修改显示逻辑。

### Q3: 如何禁用彩色输出？

**A:** 目前不支持完全禁用彩色输出。如需此功能，可以：

1. 继承 Agent 类，重写输出方法
2. 或提交 Issue 申请添加此功能

**示例解决方案**：

```python
class PlainAgent(Agent):
    """纯文本输出的 Agent"""

    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        # 保存原始方法
        self._original_print = print

    def _custom_print(self, *args, **kwargs):
        """移除颜色代码的打印"""
        text = ' '.join(str(arg) for arg in args)
        # 移除 ANSI 颜色码
        import re
        text = re.sub(r'\033\[[0-9;]*m', '', text)
        self._original_print(text, **kwargs)
```

### Q4: 如何设置工具超时？

**A:** 目前工具执行没有超时控制。如需此功能：

```python
import asyncio

class TimeoutTool(Tool):
    """带超时的工具"""

    async def execute(self, **kwargs) -> ToolResult:
        try:
            # 设置超时
            result = await asyncio.wait_for(
                self._do_work(**kwargs),
                timeout=30
            )
            return result
        except asyncio.TimeoutError:
            return ToolResult(
                success=False,
                content="",
                error="工具执行超时"
            )
```

### Q5: 如何复用 Agent 实例？

**A：** 可以安全地复用 Agent 实例：

```python
# 方式 1：重置消息历史
agent.messages = [agent.messages[0]]  # 保留系统消息

# 方式 2：重新初始化
agent.__init__(...)  # 需要确保清理状态

# 方式 3：使用工厂模式
def create_agent(config):
    return Agent(**config)
```

### Q6: 如何调试 Token 估算？

**A：** 临时添加调试信息：

```python
def debug_token_count(agent):
    """调试 Token 估算"""
    print(f"\n当前消息数量: {len(agent.messages)}")
    print(f"Token 估算: {agent._estimate_tokens()}")
    print(f"Token 限制: {agent.token_limit}")

    for i, msg in enumerate(agent.messages):
        tokens = agent._estimate_tokens_fallback()
        print(f"消息 {i}: {msg.role} - {len(msg.content)} 字符")
```

### Q7: 如何扩展消息类型？

**A：** Agent 目前只支持 system/user/assistant/tool 四种消息类型。如需扩展：

1. 修改 `Message` 数据模型（在 src/schema.py 中）
2. 更新消息处理逻辑
3. 确保向后兼容

### Q8: 支持流式输出吗？

**A：** 当前版本不支持流式输出。响应是完整返回的。如需此功能：

1. LLM 客户端需要支持流式
2. Agent 需要相应修改 `run()` 方法
3. 可以在 Issue 中申请此功能
