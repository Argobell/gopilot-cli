## 创建自定义工具

### 基本步骤

创建自定义工具只需 5 个步骤：

```mermaid
graph LR
    A[1. 继承 Tool] --> B[2. 实现 name]
    B --> C[3. 实现 description]
    C --> D[4. 实现 parameters]
    D --> E[5. 实现 execute]
```

### 步骤 1：创建工具类

在 `src/tools/` 目录下创建新文件，例如 `my_tool.py`：

```python
from typing import Any
from .base import Tool, ToolResult


class MyCustomTool(Tool):
    """我的自定义工具"""

    @property
    def name(self) -> str:
        return "my_custom_tool"

    @property
    def description(self) -> str:
        return "这是一个自定义工具的示例"

    @property
    def parameters(self) -> dict[str, Any]:
        return {
            "type": "object",
            "properties": {
                "input_text": {
                    "type": "string",
                    "description": "要处理的输入文本",
                },
                "uppercase": {
                    "type": "boolean",
                    "description": "是否转换为大写",
                    "default": False,
                },
            },
            "required": ["input_text"],
        }

    async def execute(
        self,
        input_text: str,
        uppercase: bool = False,
    ) -> ToolResult:
        """执行工具逻辑"""
        try:
            # 处理输入
            result = input_text.upper() if uppercase else input_text

            # 返回成功结果
            return ToolResult(
                success=True,
                content=result,
                error=None,
            )

        except Exception as e:
            # 返回错误结果
            return ToolResult(
                success=False,
                content="",
                error=f"处理失败: {str(e)}",
            )
```

### 步骤 2：注册工具

在 `src/tools/__init__.py` 中导出新工具：

```python
from .base import Tool, ToolResult
from .bash_tool import BashTool, BashOutputTool, BashKillTool
from .my_tool import MyCustomTool  # 添加这行

__all__ = [
    "Tool",
    "ToolResult",
    "BashTool",
    "BashOutputTool",
    "BashKillTool",
    "MyCustomTool",  # 添加这行
]
```

### 步骤 3：使用新工具

```python
from src.tools import MyCustomTool
import asyncio

async def main():
    tool = MyCustomTool()

    # 使用工具
    result = await tool.execute(
        input_text="hello world",
        uppercase=True
    )

    if result.success:
        print(result.content)  # HELLO WORLD

asyncio.run(main())
```

## 完整示例：文件操作工具

### 需求

创建一个文件读取工具，支持：
- 读取文件内容
- 可选的行数限制
- 编码检测

### 实现

```python
from pathlib import Path
from typing import Any
from .base import Tool, ToolResult


class FileReadTool(Tool):
    """文件读取工具"""

    @property
    def name(self) -> str:
        return "file_read"

    @property
    def description(self) -> str:
        return """读取文件内容的工具。

        支持：
        - 读取整个文件或指定行数
        - 自动检测编码（UTF-8、GBK）
        - 错误处理
        """

    @property
    def parameters(self) -> dict[str, Any]:
        return {
            "type": "object",
            "properties": {
                "file_path": {
                    "type": "string",
                    "description": "要读取的文件路径",
                },
                "max_lines": {
                    "type": "integer",
                    "description": "最多读取的行数（0 表示全部）",
                    "default": 0,
                },
                "encoding": {
                    "type": "string",
                    "description": "文件编码（auto 自动检测）",
                    "default": "auto",
                },
            },
            "required": ["file_path"],
        }

    async def execute(
        self,
        file_path: str,
        max_lines: int = 0,
        encoding: str = "auto",
    ) -> ToolResult:
        """读取文件内容"""
        try:
            path = Path(file_path)

            # 验证文件存在
            if not path.exists():
                return ToolResult(
                    success=False,
                    content="",
                    error=f"文件不存在: {file_path}",
                )

            # 验证是文件而非目录
            if not path.is_file():
                return ToolResult(
                    success=False,
                    content="",
                    error=f"路径不是文件: {file_path}",
                )

            # 自动检测编码
            if encoding == "auto":
                encoding = self._detect_encoding(path)

            # 读取文件
            with open(path, "r", encoding=encoding) as f:
                if max_lines > 0:
                    lines = [f.readline() for _ in range(max_lines)]
                    content = "".join(lines)
                else:
                    content = f.read()

            # 返回结果
            return ToolResult(
                success=True,
                content=content,
                error=None,
            )

        except UnicodeDecodeError as e:
            return ToolResult(
                success=False,
                content="",
                error=f"编码错误: {str(e)}",
            )

        except Exception as e:
            return ToolResult(
                success=False,
                content="",
                error=f"读取失败: {str(e)}",
            )

    def _detect_encoding(self, path: Path) -> str:
        """检测文件编码"""
        encodings = ["utf-8", "gbk", "gb2312"]

        for enc in encodings:
            try:
                with open(path, "r", encoding=enc) as f:
                    f.read()
                return enc
            except UnicodeDecodeError:
                continue

        # 默认使用 UTF-8
        return "utf-8"
```

## 错误处理策略

### 分层错误处理

```python
class RobustTool(Tool):
    async def execute(self, param: str) -> ToolResult:
        try:
            # 第一层：参数验证
            validated_param = self._validate_param(param)

            # 第二层：执行核心逻辑
            result = await self._process(validated_param)

            # 第三层：结果验证
            validated_result = self._validate_result(result)

            return ToolResult(success=True, content=validated_result)

        except ValueError as e:
            # 参数错误
            return ToolResult(
                success=False,
                content="",
                error=f"参数错误: {str(e)}",
            )

        except ConnectionError as e:
            # 连接错误
            return ToolResult(
                success=False,
                content="",
                error=f"连接失败: {str(e)}",
            )

        except Exception as e:
            # 未知错误
            return ToolResult(
                success=False,
                content="",
                error=f"未知错误: {str(e)}",
            )

    def _validate_param(self, param: str) -> str:
        """验证参数"""
        if not param:
            raise ValueError("参数不能为空")
        return param.strip()

    async def _process(self, param: str) -> str:
        """核心处理逻辑"""
        # 实现具体逻辑
        ...

    def _validate_result(self, result: str) -> str:
        """验证结果"""
        if not result:
            raise ValueError("处理结果为空")
        return result
```

### 重试机制

```python
import asyncio

class RetryableTool(Tool):
    MAX_RETRIES = 3
    RETRY_DELAY = 1  # 秒

    async def execute(self, url: str) -> ToolResult:
        for attempt in range(self.MAX_RETRIES):
            try:
                result = await self._fetch_url(url)
                return ToolResult(success=True, content=result)

            except ConnectionError as e:
                if attempt < self.MAX_RETRIES - 1:
                    # 不是最后一次尝试，等待后重试
                    await asyncio.sleep(self.RETRY_DELAY)
                    continue
                else:
                    # 最后一次尝试也失败
                    return ToolResult(
                        success=False,
                        error=f"重试 {self.MAX_RETRIES} 次后仍然失败: {str(e)}",
                        content="",
                    )

    async def _fetch_url(self, url: str) -> str:
        """获取 URL 内容"""
```

```python
            await self._process_line(line)
            processed_lines += 1

    return ToolResult(
        success=True,
        content=f"处理了 {processed_lines} 行"
    )
```

### 工具组合

组合多个工具实现复杂功能：

```python
class ComposedTool(Tool):
    def __init__(self):
        self.file_tool = FileReadTool()
        self.process_tool = ProcessTool()

    async def execute(self, file_path: str) -> ToolResult:
        # 1. 读取文件
        read_result = await self.file_tool.execute(file_path=file_path)

        # 2. 处理内容

            data=read_result.content

        return process_result
```

## 常见问题

### Q: 如何处理大文件？

**A:** 使用流式处理或分块读取，避免一次性加载到内存。

### Q: 工具执行失败时应该抛出异常还是返回错误结果？

**A:** 返回错误结果（`ToolResult(success=False, error=...)`），不要抛出异常。这样调用者可以统一处理。

### Q: 如何实现进度回调？

**A:** 可以在工具中添加可选的回调参数：

```python
from typing import Callable, Optional

async def execute(
    self,
    items: list,
    progress_callback: Optional[Callable[[int, int], None]] = None
) -> ToolResult:
    total = len(items)

    for i, item in enumerate(items):
        await self._process_item(item)

        # 调用进度回调
        if progress_callback:
            progress_callback(i + 1, total)

    return ToolResult(success=True, content="完成")
```

### Q: 如何实现超时控制？

**A:** 使用 `asyncio.wait_for()`：

```python
import asyncio

async def execute(self, task: str, timeout: int = 30) -> ToolResult:
    try:
        result = await asyncio.wait_for(
            self._long_running_task(task),
            timeout=timeout
        )
        return ToolResult(success=True, content=result)

    except asyncio.TimeoutError:
        return ToolResult(
            success=False,
            error=f"任务超时（{timeout}秒）",
            content=""
        )
```
