## 基本概念

### 什么是 Tool？

**Tool（工具）** 是一个抽象基类，定义了所有工具的通用接口。每个工具都：
- 有一个唯一的 **名称**（name）
- 提供 **描述**（description）
- 定义 **参数模式**（parameters）
- 实现 **执行方法**（execute）

### 什么是 ToolResult？

**ToolResult** 是工具执行的返回结果，包含：
- `success`：执行是否成功（bool）
- `content`：格式化后的输出内容（str）
- `error`：错误信息，如果有的话（str | None）

### 当前可用的工具

| 工具名称 | 功能 | 用途 |
|---------|------|------|
| `BashTool` | 执行 Bash/PowerShell 命令 | 前台或后台运行命令 |
| `BashOutputTool` | 监控后台进程输出 | 获取后台命令的实时输出 |
| `BashKillTool` | 终止后台进程 | 停止正在运行的后台命令 |

## 基本使用

### 示例 1：执行简单命令

```python
import asyncio
from src.tools.bash_tool import BashTool

async def main():
    # 创建工具实例
    bash_tool = BashTool()

    # 执行命令
    result = await bash_tool.execute("echo 'Hello, World!'")

    # 检查结果
    if result.success:
        print(f"输出: {result.stdout}")
        print(f"退出码: {result.exit_code}")
    else:
        print(f"错误: {result.error}")

# 运行
asyncio.run(main())
```

**输出：**
```
输出: Hello, World!
退出码: 0
```

### 示例 2：获取当前目录

```python
async def get_current_directory():
    bash_tool = BashTool()

    # Unix/Linux/macOS
    result = await bash_tool.execute("pwd")

    # Windows
    # result = await bash_tool.execute("cd")

    if result.success:
        print(f"当前目录: {result.stdout.strip()}")

    return result

asyncio.run(get_current_directory())
```

### 示例 3：处理命令错误

```python
async def handle_command_error():
    bash_tool = BashTool()

    # 执行一个会失败的命令
    result = await bash_tool.execute("exit 1")

    if not result.success:
        print(f"命令失败！")
        print(f"退出码: {result.exit_code}")
        print(f"错误信息: {result.error}")
        if result.stderr:
            print(f"错误输出: {result.stderr}")

asyncio.run(handle_command_error())
```

**输出：**
```
命令失败！
退出码: 1
错误信息: Command failed with exit code 1
```

### 示例 4：设置超时

```python
async def execute_with_timeout():
    bash_tool = BashTool()

    # 设置 10 秒超时（默认 120 秒，最大 600 秒）
    result = await bash_tool.execute(
        command="echo 'Quick task'",
        timeout=10
    )

    print(f"结果: {result.stdout}")

asyncio.run(execute_with_timeout())
```

## 后台执行

### 为什么需要后台执行？

后台执行适用于：
- 🚀 **长时间运行的服务**（如 Web 服务器）
- 📊 **需要持续监控的任务**
- 🔄 **不阻塞主程序的操作**

### 示例 5：启动后台进程

```python
from src.tools.bash_tool import BashTool, BashOutputTool, BashKillTool
import asyncio

async def start_background_process():
    bash_tool = BashTool()

    # 启动后台 HTTP 服务器
    result = await bash_tool.execute(
        command="python -m http.server 8080",
        run_in_background=True
    )

    if result.success:
        print(f"服务已启动！")
        print(f"进程 ID: {result.bash_id}")
        return result.bash_id
    else:
        print(f"启动失败: {result.error}")
        return None

bash_id = asyncio.run(start_background_process())
```

**输出：**
```
服务已启动！
进程 ID: abc12345
```

### 示例 6：监控后台输出

### 示例 7：过滤输出

```python
async def filter_error_output(bash_id):
    output_tool = BashOutputTool()

    # 只获取包含 "ERROR" 的输出行
    result = await output_tool.execute(
        bash_id=bash_id,

    else:
```

### 示例 9：完整的后台任务管理

```python
async def complete_background_task_example():
    """完整示例：启动、监控、终止后台任务"""

    bash_tool = BashTool()
    output_tool = BashOutputTool()
    kill_tool = BashKillTool()

    # 1. 启动后台进程
    print("1️⃣ 启动后台进程...")
    start_result = await bash_tool.execute(
        command="python -c 'import time; [print(f\"Running {i}\") or time.sleep(1) for i in range(10)]'",
        run_in_background=True
    )

    if not start_result.success:
        print(f"启动失败: {start_result.error}")
        return

    bash_id = start_result.bash_id
    print(f"✅ 进程已启动，ID: {bash_id}")

    # 2. 监控输出（循环 3 次）
    print("\n2️⃣ 监控输出...")
    for i in range(3):
        await asyncio.sleep(2)
        output_result = await output_tool.execute(bash_id=bash_id)

        if output_result.success:
            print(f"--- 第 {i+1} 次检查 ---")
            print(output_result.stdout or "(暂无新输出)")

    # 3. 终止进程
    print("\n3️⃣ 终止进程...")
    kill_result = await kill_tool.execute(bash_id=bash_id)

    if kill_result.success:
        print("✅ 进程已终止")
        if kill_result.stdout:
            print("最后输出:")
            print(kill_result.stdout)
    else:
        print(f"❌ 终止失败: {kill_result.error}")

# 运行完整示例
asyncio.run(complete_background_task_example())
```

## 常见使用场景

### 场景 1：Git 操作

```python
async def git_operations():
    bash_tool = BashTool()

    # 检查 Git 状态
    status = await bash_tool.execute("git status")
    print(status.stdout)

    # 查看最近提交
    log = await bash_tool.execute("git log --oneline -5")
    print(log.stdout)

    # 检查分支
    branch = await bash_tool.execute("git branch")
    print(branch.stdout)

asyncio.run(git_operations())
```

### 场景 4：启动开发服务器

```python
async def start_dev_server():
    bash_tool = BashTool()
    output_tool = BashOutputTool()

    # 启动 FastAPI 开发服务器
    result = await bash_tool.execute(
        command="uvicorn main:app --reload --port 8000",

        return bash_id

    return None
```

### 场景 5：批量处理

```python
async def batch_processing():
    bash_tool = BashTool()

    commands = [
        "echo 'Task 1'",
        "echo 'Task 2'",
        "echo 'Task 3'",
    ]

    print("批量执行任务...")
    results = []

    for i, cmd in enumerate(commands, 1):
        print(f"执行任务 {i}...")
        result = await bash_tool.execute(cmd)
        results.append(result)

        if result.success:
            print(f"  ✅ {result.stdout.strip()}")
        else:
            print(f"  ❌ {result.error}")

    # 统计
    success_count = sum(1 for r in results if r.success)
    print(f"\n完成：{success_count}/{len(commands)} 个任务成功")

asyncio.run(batch_processing())
```

### 场景 6：并发执行

```python
async def concurrent_execution():
    bash_tool = BashTool()

    # 同时执行多个命令
    tasks = [
        bash_tool.execute("echo 'Task A'"),
        bash_tool.execute("echo 'Task B'"),
        bash_tool.execute("echo 'Task C'"),
    ]

    print("并发执行任务...")
    results = await asyncio.gather(*tasks)

    for i, result in enumerate(results, 1):
        if result.success:
            print(f"任务 {i}: {result.stdout.strip()}")

asyncio.run(concurrent_execution())
```

### 常见问题

#### Q: 如何处理长时间运行的命令？

**A:** 使用后台执行模式（`run_in_background=True`），并通过 `BashOutputTool` 定期检查输出。

#### Q: 命令超时了怎么办？

**A:** 可以增加 `timeout` 参数。默认是 120 秒，最大可设置为 600 秒（10 分钟）。对于更长的任务，使用后台执行。

#### Q: 如何在 Windows 上使用？

**A:** Tools 模块会自动检测操作系统。在 Windows 上，命令将通过 PowerShell 执行。确保命令语法兼容 PowerShell。

#### Q: 后台进程的输出丢失了怎么办？

**A:** `BashOutputTool` 返回的是自上次读取后的**新输出**。如果想保留所有输出，需要在应用层自己缓存。

#### Q: 如何调试工具执行？

**A:** 检查 `ToolResult` 的 `error` 字段和 `stderr` 字段。对于后台进程，使用 `BashOutputTool` 查看实时输出。
